Problem Statement- 
    Sam want to read exactly TARGET' number of pages.
    He has an array ‘BOOK’ containing the number of pages for ‘N* |
    books. |
    Return YES/NO, if it is possible for him to read any 2 books and he
    can meet his TARGET' number of pages.

    Input: ‘N’ =5, ‘TARGET’ =5
    “BOOK’ = [4, 1, 2, 3, 1]
    Output: YES
    Explanation:
    Sam can buy 4 pages book and 1 page book. 

Function: 
        #include<map>
        string read(int n, vector<int> book, int target)
        {
            // Write your code here.
            //better approach
            // map<int,int> mp;
            // for(int i=0;i<n;i++)
            // {

            //     if(mp.find(need) != mp.end())
            //         return "YES";
            //     mp[book[i]] = i;
            // }
            // return "NO";
            
            //Optimal Approach
            int left =0, right =n-1;
            sort(book.begin(), book.end());
            while(left<right)
            {
                int sum = book[left] + book[right];
                if(sum == target)
                    return "YES";
                else if (sum < target) left++;
                else right--;
            }
            return "NO";
        }
